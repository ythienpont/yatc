81 static Message parseMessage(const std::vector<std::byte> &data) {
82 if (data.size() < 5)
83 throw std::runtime_error("Invalid message data received.");
84
85 MessageType type = static_cast<MessageType>(static_cast<uint8_t>(data[4]));
86 std::vector<std::byte> payloadBytes(data.begin() + 5, data.end());
87
88 switch (type) {
89 case MessageType::Choke:
90 case MessageType::Unchoke:
91 case MessageType::Interested:
92 case MessageType::NotInterested:
93 return {type, std::monostate{}};
94
95 case MessageType::Have:
96 if (payloadBytes.size() < sizeof(uint32_t))
97 throw std::runtime_error("Invalid payload size for Have message.");
98 return {type, bytes_to_uint32(payloadBytes)};
99
100 case MessageType::Bitfield:
101 return {type, payloadBytes};
102
103 case MessageType::Request:
104 case MessageType::Cancel:
105 if (payloadBytes.size() < 3 * sizeof(uint32_t))
106 throw std::runtime_error("Invalid payload size for Request/Cancel message.");
107 return {type, std::make_tuple(bytes_to_uint32(payloadBytes, 0),
108 bytes_to_uint32(payloadBytes, 4),
109 bytes_to_uint32(payloadBytes, 8))};
110
111 case MessageType::Piece:
112 if (payloadBytes.size() < 2 * sizeof(uint32_t))
113 throw std::runtime_error("Invalid payload size for Piece message.");
114 {
115 uint32_t index = bytes_to_uint32(payloadBytes, 0);
116 uint32_t begin = bytes_to_uint32(payloadBytes, 4);
117 std::vector<std::byte> block(payloadBytes.begin() + 8, payloadBytes.end());
118 return {type, PieceData{index, begin, std::move(block)}};
119 }
120
121 default:
122 throw std::runtime_error("Unknown message type.");
123 }
